### Instructions for running robot_docker 
cd eurobin_robot 
source robot_profile.bash
cd robot_docker 
robot dev   (which calls the setup.sh and directly calls the docker compose)

(or)
cd robot_docker
docker compose build --no-cache 
docker compose up dev -d
docker compose exec dev bash


### Instructions for running perception_docker 
cd perception_docker
docker compose build --no-cache perception
docker compose up perception -d
docker compose exec perception bash
